CREATE TABLE assunto (ID  SERIAL NOT NULL, PALAVRA VARCHAR(255), VERSAO VARCHAR(255), PRIMARY KEY (ID))
CREATE TABLE usuario (ID  SERIAL NOT NULL, DATANASC TIMESTAMP, EMAIL VARCHAR(255), VERSAO VARCHAR(255), PRIMARY KEY (ID))
CREATE TABLE VISUALIZACAO (ID  SERIAL NOT NULL, DATAHORA DATETIME, NOTA INTEGER, VERSAO VARCHAR(255), USUARIO_ID INTEGER, VIDEO_ID INTEGER, PRIMARY KEY (ID))
CREATE TABLE video (ID  SERIAL NOT NULL, DATAHORA VARCHAR(255), LINK VARCHAR(255), MEDIA FLOAT, NOME VARCHAR(255), VERSAO VARCHAR(255), PRIMARY KEY (ID))
CREATE TABLE assunto_video (assuntos_ID INTEGER NOT NULL, videos_ID INTEGER NOT NULL, PRIMARY KEY (assuntos_ID, videos_ID))
ALTER TABLE VISUALIZACAO ADD CONSTRAINT FK_VISUALIZACAO_USUARIO_ID FOREIGN KEY (USUARIO_ID) REFERENCES usuario (ID)
