CREATE TABLE REUNIAO (DATAHORA TIMESTAMP, ASSUNTO VARCHAR(255), PRIMARY KEY (DATAHORA))
CREATE TABLE VIAGEM (ID  SERIAL NOT NULL, DATA DATE, DESTINO VARCHAR(255), minhachave INTEGER, PRIMARY KEY (ID))
CREATE TABLE TELEFONE (ID  SERIAL NOT NULL, NUMERO VARCHAR(255), PESSOA_ID INTEGER, PRIMARY KEY (ID))
CREATE TABLE Pessoa_APELIDOS (Pessoa_ID INTEGER, APELIDOS VARCHAR(255))
CREATE TABLE REUNIAO_PESSOA (participantes_ID INTEGER NOT NULL, reunioes_DATAHORA TIMESTAMP, PRIMARY KEY (participantes_ID, reunioes_DATAHORA))
